$.getJSON("//int.nyt.com/applications/elections/2020/data/api/2020-02-11/new-hampshire/president/democrat.json", function(json){
 jsondata = json;
});

}
 document.getElementById("demo").innerHTML += "<br/>"

  document.getElementById("demo").innerHTML += file.counties[i].fips+ "," + "IA" + "," + file.counties[i].votes + ",";

  keys = Object.keys(file.counties[i].results);

  }
