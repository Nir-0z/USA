<div id="demo"></div>

<script>
var jsonfile = [];

for (var i=0; i < jsonfile.length; i++) {
 var sortable = [];
 for (var item in jsonfile[i]) {
  sortable.push([item, jsonfile[i][item]]);
 }
sortable.sort(function(a, b) {
    return b[1] - a[1];
});
document.getElementById("demo").innerHTML += sortable + "<br/>";
}
</script>
