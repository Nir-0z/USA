<script src = "https://ajax.googleapis.com/ajax/libs/jquery/2.1.3/jquery.min.js"></script>
<body><div id="demo"></div></body>

<script>
var jsondata;
$.getJSON("//int.nyt.com/applications/elections/2020/data/api/2020-02-03/iowa/president/democrat.json", function(json){
 jsondata = json;
});

var file = jsondata.data.races[0];

document.getElementById("demo").innerHTML += "US-" + file.state_id + "," + file.state_id + "," + file.votes + ",";
for (var o = 0; o <= 13; o++) {
  document.getElementById("demo").innerHTML += file.candidates[o].candidate_key + "," + file.candidates[o].votes + "," + ",";
}
document.getElementById("demo").innerHTML += "<br/>";

for (var i = 0; i <= 999999; i++) {
  document.getElementById("demo").innerHTML += file.counties[i].fips+ "," + "IA" + "," + file.counties[i].votes_alignfinal + ",";
  var subfile = file.counties[i].results_alignfinal;
  keys = Object.keys(file.counties[i].results_alignfinal);
  for (var j = 0; j < Object.keys(file.counties[i].results).length; j++) {
   document.getElementById("demo").innerHTML += keys[j] + "," + subfile[keys[j]] + "," + ",";
  }
  document.getElementById("demo").innerHTML += "<br/>" 
}
</script>
